function isInboundUrl(url) {
var inboundRegex = [
/\/\/[^\/]*komica.org[\/]/g,
/\/\/[^\/]*komica2.net[\/]/g
];
for (var i=0; i<inboundRegex.length; i++) {
if (inboundRegex[i].test(url))
return true;
}
return false;
}
function escapeRegExp(str) {
return str.replace(/[\-\[\]\/\{\}\(\)\*\+\?\.\\\^\$\|]/g, "\\$&");
}

var Bookmark = (function(){
function Bookmark() {
if (typeof localStorage !== 'undefined') {
localStorage = [];
}
this.list = [];
this.invList = {}; //inverse function of list
this.load();
}
Bookmark.prototype.arraySwap = function(ar, x, y) {
var tmp = ar[x];
ar[x] = ar[y];
ar[y] = tmp;
}
Bookmark.prototype.load = function() {
var txt = '[]';
var result = [];
try {
txt = localStorage['bookmarkK1'];
} catch(e) {
txt = '[]';
}
try {
result = JSON.parse(txt);
} catch(e) {
result = [];
}
if (!Array.isArray(result)) {
result = [];
}
this.list = [];
this.invList = {};
for (var i=0; i<result.length; i++) {
var item = result[i];
this.list.push(item);
this.invList[item] = this.list.length - 1;
}
}
Bookmark.prototype.save = function() {
try {
localStorage['bookmarkK1'] = JSON.stringify(this.list);
} catch(e) {}
}
Bookmark.prototype.moveUp = function(board) {
var pos = this.invList[board];
if (pos == 0) return;
var board2 = this.list[pos-1];
this.arraySwap(this.list, pos-1, pos);
this.arraySwap(this.invList, board, board2);
this.save();
}
Bookmark.prototype.moveDown = function(board) {
var pos = this.invList[board];
if (pos >= this.list.length - 1) return;
this.moveUp( this.list[pos+1] );
}
Bookmark.prototype.getBoardElement = function(board) {
var result = null;
$('#list ul:not(#bookmark) li:not(.category)').each(function(){
var txt = $(this).find('a').html();
if (result===null && txt==board)
result = $(this);
});
return result;
}
Bookmark.prototype.hasBoard = function(board) {
var pos = this.invList[board];
if (pos===null || pos===undefined)
return false;
return true;
}
Bookmark.prototype.addBoard = function(board) {
if (this.hasBoard(board)) return;
this.list.push(board);
this.invList[board] = this.list.length - 1;
this.save();
}
Bookmark.prototype.removeBoard = function(board) {
var pos = -1;
for (var i=0; i<this.list.length; i++) {
if (this.list[i] == board) {
pos = i;
break;
}
}
if (pos == -1) return;
for (var i=pos; i<=this.list.length-2; i++) {
this.list[i] = this.list[i+1];
this.invList[this.list[i]] = i;
}
this.list.pop();
delete this.invList[board];
this.save();
}
return Bookmark;
})();

}

}
